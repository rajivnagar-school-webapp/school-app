STEP 2 - SUPABASE SETUP
------------------------
supabase.com > New Project > SQL Editor:

CREATE TABLE IF NOT EXISTS users (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    name TEXT NOT NULL,
    username TEXT UNIQUE NOT NULL,
    password_hash TEXT NOT NULL,
    role TEXT NOT NULL CHECK (role IN ('teacher','principal')),
    class_assigned TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS students (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    name TEXT NOT NULL,
    surname TEXT,
    father_name TEXT,
    mother_name TEXT,
    roll_no TEXT NOT NULL,
    gr_number TEXT,
    class TEXT NOT NULL,
    dob TEXT,
    gender TEXT,
    caste TEXT,
    section TEXT,
    address TEXT,
    parent_contact TEXT,
    aadhaar_number TEXT,
    bank_account TEXT,
    bank_name TEXT,
    ifsc_code TEXT,
    attendance_register_no TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS student_documents (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    student_id UUID REFERENCES students(id) ON DELETE CASCADE,
    doc_name TEXT NOT NULL,
    doc_url TEXT NOT NULL,
    file_ext TEXT,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS attendance (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    student_id UUID REFERENCES students(id) ON DELETE CASCADE,
    date DATE NOT NULL,
    status TEXT NOT NULL CHECK (status IN ('P','A')),
    marked_by UUID,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS subjects (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    class TEXT NOT NULL,
    subject_name TEXT NOT NULL,
    teacher_id UUID,
    max_written INTEGER DEFAULT 40,
    max_participation INTEGER DEFAULT 20,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS marks (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    student_id UUID REFERENCES students(id) ON DELETE CASCADE,
    subject_id UUID REFERENCES subjects(id) ON DELETE CASCADE,
    semester TEXT NOT NULL,
    written_marks INTEGER DEFAULT 0,
    participation_marks INTEGER DEFAULT 0,
    academic_year TEXT DEFAULT '2025-26',
    entered_by UUID,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

ALTER TABLE users             DISABLE ROW LEVEL SECURITY;
ALTER TABLE students          DISABLE ROW LEVEL SECURITY;
ALTER TABLE student_documents DISABLE ROW LEVEL SECURITY;
ALTER TABLE attendance        DISABLE ROW LEVEL SECURITY;
ALTER TABLE subjects          DISABLE ROW LEVEL SECURITY;
ALTER TABLE marks             DISABLE ROW LEVEL SECURITY;

ALTER TABLE students ADD COLUMN IF NOT EXISTS address TEXT;
ALTER TABLE students ADD COLUMN IF NOT EXISTS bank_name TEXT;
ALTER TABLE students ADD COLUMN IF NOT EXISTS ifsc_code TEXT;
ALTER TABLE students ADD COLUMN IF NOT EXISTS attendance_register_no TEXT;
ALTER TABLE student_documents ADD COLUMN IF NOT EXISTS file_ext TEXT;
ALTER TABLE subjects ADD COLUMN IF NOT EXISTS max_written INTEGER DEFAULT 40;
ALTER TABLE subjects ADD COLUMN IF NOT EXISTS max_participation INTEGER DEFAULT 20;
